### **61. Firestore — Data Model, Indexing & Production Patterns**

> **Level:** Advanced

#### Why This Is Asked

HighLevel's tech stack explicitly lists Firestore. Interviewers want to know whether you understand its document-collection model, its different indexing rules from MongoDB, its transaction model, and — critically — when to choose it over MongoDB and how to stream data out of it into ClickHouse for analytics.

#### Explanation

**Document-Collection Model**

Firestore stores data as **documents** (JSON-like) inside **collections**. Collections can contain **sub-collections** (e.g., `contacts/{contactId}/activities`). There are no joins — you denormalise or use collection-group queries.

**Indexing Rules**

- Every field gets a **single-field index** automatically.
- Queries that combine a range filter (`>`, `<`, `>=`, `<=`) on one field **and** filter/sort on another field require a **composite index** (defined explicitly in `firestore.indexes.json`).
- Queries with two range filters on different fields are not supported.
- The `!=` operator and `not-in` operator both require composite indexes.

**Transactions vs Batch Writes**

| | Transaction | Batch Write |
|---|---|---|
| Reads + Writes | Yes | Writes only |
| Atomicity | Yes (all or nothing) | Yes |
| Max ops | 500 | 500 |
| Retries on contention | Automatic | No |
| Use when | Read-modify-write | Pure multi-doc writes |

**Firestore vs MongoDB**

| | Firestore | MongoDB |
|---|---|---|
| Scaling | Serverless, auto-scales | Manual sharding |
| Query flexibility | Limited (no aggregations) | Rich (`$group`, `$lookup`) |
| Real-time | Built-in `onSnapshot` | Change streams (manual) |
| Cost model | Per-read/write operation | Per instance |
| Best for | Operational data needing real-time sync | Complex queries, analytics |

**Firestore → ClickHouse Streaming**

Use Firestore's `onSnapshot` listener (or Cloud Firestore Change Events via Eventarc) → publish to Pub/Sub → ClickHouse consumer. This is the canonical HighLevel analytics pipeline.

#### Code Example

```javascript
const { Firestore, FieldValue } = require('@google-cloud/firestore');
const db = new Firestore({ projectId: process.env.GCP_PROJECT });

// ── Basic CRUD with tenant isolation ─────────────────────────────────────
async function createContact(tenantId, contact) {
  const ref = db
    .collection('tenants').doc(tenantId)
    .collection('contacts').doc(); // auto-generated ID

  await ref.set({
    ...contact,
    createdAt: FieldValue.serverTimestamp(),
    updatedAt: FieldValue.serverTimestamp(),
  });
  return ref.id;
}

async function updateContact(tenantId, contactId, patch) {
  const ref = db
    .collection('tenants').doc(tenantId)
    .collection('contacts').doc(contactId);

  await ref.update({
    ...patch,
    updatedAt: FieldValue.serverTimestamp(),
  });
}

// ── Transaction — read-modify-write ───────────────────────────────────────
async function incrementDealCount(tenantId, contactId) {
  const ref = db
    .collection('tenants').doc(tenantId)
    .collection('contacts').doc(contactId);

  await db.runTransaction(async (tx) => {
    const doc = await tx.get(ref);
    if (!doc.exists) throw new Error('Contact not found');
    tx.update(ref, { dealCount: (doc.data().dealCount ?? 0) + 1 });
  });
}

// ── Batch write — atomic multi-document writes ────────────────────────────
async function bulkCreateActivities(tenantId, activities) {
  // Max 500 per batch
  const chunks = [];
  for (let i = 0; i < activities.length; i += 500) {
    chunks.push(activities.slice(i, i + 500));
  }

  for (const chunk of chunks) {
    const batch = db.batch();
    for (const activity of chunk) {
      const ref = db
        .collection('tenants').doc(tenantId)
        .collection('activities').doc();
      batch.set(ref, { ...activity, createdAt: FieldValue.serverTimestamp() });
    }
    await batch.commit();
  }
}

// ── Real-time listener → Pub/Sub pipeline ─────────────────────────────────
const { PubSub } = require('@google-cloud/pubsub');
const pubsub = new PubSub({ projectId: process.env.GCP_PROJECT });

function startFirestoreChangePipeline(tenantId) {
  const topic = pubsub.topic('firestore-changes');

  db.collection('tenants').doc(tenantId)
    .collection('contacts')
    .onSnapshot(
      { includeMetadataChanges: false },
      async (snapshot) => {
        const changes = snapshot.docChanges().map(change => ({
          type: change.type,       // 'added' | 'modified' | 'removed'
          id: change.doc.id,
          tenantId,
          data: change.doc.data(),
          timestamp: new Date().toISOString(),
        }));

        if (!changes.length) return;

        await topic.publishMessage({
          json: { changes },
          orderingKey: tenantId, // preserve per-tenant order
        });
      },
      (err) => console.error('Snapshot listener error', err)
    );
}

// ── Collection-group query (across all tenants' contacts sub-collections) ──
async function findContactsByEmail(email) {
  // Requires composite index: __name__ asc + email asc
  const snapshot = await db
    .collectionGroup('contacts')
    .where('email', '==', email)
    .limit(10)
    .get();

  return snapshot.docs.map(d => ({ id: d.id, ...d.data() }));
}

// ── Paginated list with cursor ─────────────────────────────────────────────
async function listContacts(tenantId, { limit = 50, startAfterDoc = null } = {}) {
  let query = db
    .collection('tenants').doc(tenantId)
    .collection('contacts')
    .orderBy('createdAt', 'desc')
    .limit(limit);

  if (startAfterDoc) query = query.startAfter(startAfterDoc);

  const snapshot = await query.get();
  const docs = snapshot.docs.map(d => ({ id: d.id, ...d.data() }));
  const nextCursor = snapshot.docs[snapshot.docs.length - 1] ?? null;
  return { docs, nextCursor };
}
```

#### Do's and Don'ts

**Do's**
- Use `FieldValue.serverTimestamp()` instead of `new Date()` for write timestamps — it avoids client clock skew
- Define composite indexes in `firestore.indexes.json` and deploy them before queries need them (index creation is async and takes minutes)
- Use cursors (`startAfter`) for pagination — offset-based pagination (`offset(n)`) reads and discards documents, which costs money and time
- Batch writes for bulk inserts (up to 500 per batch) to reduce round-trips
- Scope all queries under `tenants/{tenantId}/...` paths for natural tenant isolation

**Don'ts**
- Don't use Firestore for complex aggregation queries — use ClickHouse or a read model in MongoDB
- Don't open unbounded `onSnapshot` listeners in API handlers — they leak if the handler is called repeatedly without cleanup
- Don't put large arrays or deeply nested objects in documents — Firestore charges per field read
- Don't rely on Firestore security rules as your only auth mechanism in server-side code; always validate tenantId in application code
- Don't use `offset(n)` for pagination in high-document collections — it gets slower as n grows

#### Common Follow-up Questions

1. How does Firestore's `onSnapshot` listener behave when the network drops? Does it re-emit buffered changes on reconnect?
2. Firestore has a 1 write/second limit per document. How do you handle a counter that is incremented by thousands of events per second?
3. How would you migrate 10 million Firestore documents to a new schema without downtime?

### **62. Redis Caching Strategies — Deep Dive**

> **Level:** Advanced

#### Why This Is Asked

Redis is listed as a required technology. Interviewers want more than "I use Redis for caching" — they probe whether you know the difference between cache patterns, how to handle cache stampedes, which eviction policy fits which workload, and how to design a multi-tier cache for a multi-tenant SaaS.

#### Explanation

**Cache Patterns**

| Pattern | How it works | Write latency | Consistency |
|---|---|---|---|
| **Cache-aside (lazy)** | App checks cache; on miss, reads DB, populates cache | Baseline | Eventual (TTL-based) |
| **Write-through** | App writes to cache AND DB synchronously | Higher | Strong |
| **Write-behind (write-back)** | App writes to cache; async job flushes to DB | Low | Eventual |
| **Read-through** | Cache fetches from DB on miss automatically | Baseline | Eventual |
| **Refresh-ahead** | Cache proactively refreshes before TTL expires | None (background) | Near-real-time |

**Cache Stampede (Thundering Herd)**

When a hot key expires simultaneously, all concurrent requests hit the DB at once. Solutions:
1. **Mutex lock** — first request acquires a Redis lock, others wait or return stale data
2. **Probabilistic early expiry** (PER) — slightly before TTL, randomly decide to refresh early
3. **Stale-while-revalidate** — serve stale data immediately, refresh in background

**Eviction Policies**

| Policy | Evicts | Best for |
|---|---|---|
| `allkeys-lru` | Least recently used (any key) | General caching |
| `volatile-lru` | LRU keys with TTL set | Mix of cache + persistent data |
| `allkeys-lfu` | Least frequently used | Hot/cold access patterns |
| `volatile-ttl` | Key with nearest expiry | Time-sensitive data |
| `noeviction` | Throws error when full | Session store — must not lose data |

**Multi-Tier Caching**

L1: in-process memory (Map or `lru-cache`) — sub-millisecond, bounded size, per-process
L2: Redis — shared across all instances, milliseconds
L3: Database — source of truth

#### Code Example

```javascript
const redis = require('ioredis');
const { LRUCache } = require('lru-cache');

const redisClient = new redis(process.env.REDIS_URL);

// ── L1 in-process + L2 Redis multi-tier cache ─────────────────────────────
const l1Cache = new LRUCache({ max: 1000, ttl: 5_000 }); // 5 s, max 1000 entries

async function getContact(tenantId, contactId) {
  const key = `contact:${tenantId}:${contactId}`;

  // L1 hit
  const l1 = l1Cache.get(key);
  if (l1) return l1;

  // L2 hit
  const l2 = await redisClient.get(key);
  if (l2) {
    const parsed = JSON.parse(l2);
    l1Cache.set(key, parsed);
    return parsed;
  }

  // L3 — DB
  const contact = await db.collection('contacts').findOne({ _id: contactId, tenantId });
  if (contact) {
    const serialised = JSON.stringify(contact);
    await redisClient.set(key, serialised, 'EX', 300); // 5 min TTL
    l1Cache.set(key, contact);
  }
  return contact;
}

// ── Cache invalidation on write ────────────────────────────────────────────
async function updateContact(tenantId, contactId, patch) {
  await db.collection('contacts').updateOne(
    { _id: contactId, tenantId },
    { $set: { ...patch, updatedAt: new Date() } }
  );

  // Invalidate both tiers
  const key = `contact:${tenantId}:${contactId}`;
  l1Cache.delete(key);
  await redisClient.del(key);

  // Also invalidate list caches for this tenant
  const listKeys = await redisClient.keys(`contacts:list:${tenantId}:*`);
  if (listKeys.length) await redisClient.del(...listKeys);
}

// ── Mutex lock to prevent cache stampede ──────────────────────────────────
async function getWithMutex(key, fetchFn, ttl = 300) {
  const cached = await redisClient.get(key);
  if (cached) return JSON.parse(cached);

  const lockKey = `lock:${key}`;
  const lockToken = crypto.randomUUID();

  // Try to acquire lock (SET NX with 10s expiry)
  const acquired = await redisClient.set(lockKey, lockToken, 'EX', 10, 'NX');

  if (!acquired) {
    // Another instance is fetching — wait and retry
    await new Promise(res => setTimeout(res, 100));
    const retried = await redisClient.get(key);
    return retried ? JSON.parse(retried) : getWithMutex(key, fetchFn, ttl);
  }

  try {
    const data = await fetchFn();
    await redisClient.set(key, JSON.stringify(data), 'EX', ttl);
    return data;
  } finally {
    // Release lock only if we still own it
    const script = `if redis.call('get', KEYS[1]) == ARGV[1] then return redis.call('del', KEYS[1]) else return 0 end`;
    await redisClient.eval(script, 1, lockKey, lockToken);
  }
}

// ── Stale-while-revalidate ─────────────────────────────────────────────────
async function getStaleWhileRevalidate(key, fetchFn, { ttl = 300, staleTtl = 30 } = {}) {
  const raw = await redisClient.get(key);
  if (raw) {
    const { data, expiresAt } = JSON.parse(raw);
    // If within stale window, return now and refresh in background
    if (Date.now() > expiresAt - staleTtl * 1000) {
      setImmediate(async () => {
        try {
          const fresh = await fetchFn();
          await redisClient.set(key, JSON.stringify({
            data: fresh,
            expiresAt: Date.now() + ttl * 1000,
          }), 'EX', ttl + staleTtl);
        } catch {}
      });
    }
    return data;
  }

  const data = await fetchFn();
  await redisClient.set(key, JSON.stringify({
    data,
    expiresAt: Date.now() + ttl * 1000,
  }), 'EX', ttl + staleTtl);
  return data;
}

// ── Write-through cache ────────────────────────────────────────────────────
async function writeThroughUpdateContact(tenantId, contactId, patch) {
  const key = `contact:${tenantId}:${contactId}`;

  // Write to DB first
  const updated = await db.collection('contacts').findOneAndUpdate(
    { _id: contactId, tenantId },
    { $set: { ...patch, updatedAt: new Date() } },
    { returnDocument: 'after' }
  );

  // Immediately update cache (write-through)
  if (updated.value) {
    await redisClient.set(key, JSON.stringify(updated.value), 'EX', 300);
    l1Cache.set(key, updated.value);
  }

  return updated.value;
}
```

#### Do's and Don'ts

**Do's**
- Always scope cache keys by tenantId to prevent cross-tenant data leakage
- Use mutex locks or stale-while-revalidate for hot keys to prevent stampedes
- Set `maxmemory-policy allkeys-lru` in Redis for pure cache workloads — avoid OOM errors
- Use `SCAN` instead of `KEYS` in production to find keys matching a pattern (KEYS blocks the event loop)
- Version your cache keys (`contact:v2:tenantId:id`) when the cached schema changes

**Don'ts**
- Don't cache mutable lists without a clear invalidation strategy — list caches go stale silently
- Don't use `KEYS *` in production — it blocks Redis for the duration of the scan on large datasets
- Don't set the same TTL for all data types — hot, frequently-changing data should have a shorter TTL
- Don't store objects without serialisation versioning — a schema change in a cached object can cause runtime errors when deserialising old cache entries
- Don't use write-behind caching for financial or compliance-critical data — a crash before the flush causes data loss

#### Common Follow-up Questions

1. How would you implement cache warming after a Redis restart to prevent a cache miss storm?
2. What is a "dog-pile effect" and how does probabilistic early expiry solve it differently from a mutex?
3. How do you monitor cache hit ratio and eviction rate in Prometheus, and what thresholds would trigger an alert?

### **63. Rate Limiting Algorithms & Distributed Implementation**

> **Level:** Advanced

#### Why This Is Asked

HighLevel handles billions of API hits. Rate limiting is the primary mechanism for protecting services from abuse, controlling noisy tenants, and ensuring fair use. A Lead Engineer must know which algorithm fits which use case and how to implement them correctly in a distributed cluster using Redis.

#### Explanation

**Fixed Window Counter**

Count requests in a fixed time window (e.g., each minute). Simple but has a burst problem: a client can send 2× the limit by straddling two windows.

**Sliding Window Log**

Store a timestamped log of each request. Count log entries in the last N seconds. Accurate but memory-intensive (O(requests) per client).

**Sliding Window Counter**

Approximate sliding window by weighting the current window count with the fraction of the previous window still in scope. Memory-efficient (O(1) per client), highly accurate.

```
rate = prev_window_count × ((window_size - elapsed) / window_size) + curr_window_count
```

**Token Bucket**

A bucket holds tokens (up to `capacity`). Tokens are added at a fixed `refillRate`. Each request consumes one token. If the bucket is empty, reject the request. Allows short bursts up to bucket capacity.

**Leaky Bucket**

Requests enter a queue and are processed at a fixed rate. Excess requests are dropped. Smooths traffic but does not allow any burst.

| Algorithm | Burst allowed | Memory | Accuracy | Best for |
|---|---|---|---|---|
| Fixed window | Yes (2×) | O(1) | Low | Simple quotas |
| Sliding window log | No | O(requests) | Exact | Audit / billing |
| Sliding window counter | Small | O(1) | High | API gateways |
| Token bucket | Yes (up to capacity) | O(1) | High | API rate limiting |
| Leaky bucket | No | O(queue) | Exact | Traffic shaping |

#### Code Example

```javascript
// ── Sliding Window Counter — Redis Lua script (atomic) ────────────────────
const slidingWindowScript = `
  local key = KEYS[1]
  local now = tonumber(ARGV[1])
  local window = tonumber(ARGV[2])
  local limit = tonumber(ARGV[3])

  local prevKey = key .. ':prev'
  local currKey = key .. ':curr'
  local windowStart = now - window

  -- Get counts
  local prevCount = tonumber(redis.call('GET', prevKey) or '0')
  local currCount = tonumber(redis.call('GET', currKey) or '0')

  -- How far into the current window are we? (0.0 – 1.0)
  local elapsed = (now % window) / window

  -- Weighted count
  local rate = prevCount * (1 - elapsed) + currCount

  if rate >= limit then
    return 0  -- reject
  end

  -- Increment current window counter
  redis.call('INCR', currKey)
  redis.call('PEXPIRE', currKey, window * 2)  -- keep alive for 2 windows

  return 1  -- allow
`;

async function slidingWindowRateLimit(tenantId, endpoint, limit, windowMs) {
  const key = `rl:${tenantId}:${endpoint}`;
  const now = Date.now();

  const result = await redisClient.eval(
    slidingWindowScript,
    1,
    key,
    now,
    windowMs,
    limit
  );

  return result === 1; // true = allowed
}

// ── Token Bucket — Redis Lua script ───────────────────────────────────────
const tokenBucketScript = `
  local key = KEYS[1]
  local capacity = tonumber(ARGV[1])
  local refillRate = tonumber(ARGV[2])  -- tokens per ms
  local now = tonumber(ARGV[3])
  local cost = tonumber(ARGV[4] or '1')

  local bucket = redis.call('HMGET', key, 'tokens', 'lastRefill')
  local tokens = tonumber(bucket[1] or capacity)
  local lastRefill = tonumber(bucket[2] or now)

  -- Refill tokens based on elapsed time
  local elapsed = now - lastRefill
  tokens = math.min(capacity, tokens + elapsed * refillRate)

  if tokens < cost then
    -- Update lastRefill timestamp even on reject
    redis.call('HMSET', key, 'tokens', tokens, 'lastRefill', now)
    redis.call('PEXPIRE', key, 60000)
    return {0, math.ceil((cost - tokens) / refillRate)}  -- {rejected, retry_after_ms}
  end

  tokens = tokens - cost
  redis.call('HMSET', key, 'tokens', tokens, 'lastRefill', now)
  redis.call('PEXPIRE', key, 60000)
  return {1, 0}  -- {allowed, 0}
`;

async function tokenBucketRateLimit(tenantId, { capacity, refillRatePerSec, cost = 1 }) {
  const key = `tb:${tenantId}`;
  const refillRatePerMs = refillRatePerSec / 1000;

  const [allowed, retryAfterMs] = await redisClient.eval(
    tokenBucketScript,
    1,
    key,
    capacity,
    refillRatePerMs,
    Date.now(),
    cost
  );

  return { allowed: allowed === 1, retryAfterMs };
}

// ── Express middleware combining plan-based limits ─────────────────────────
const PLAN_LIMITS = {
  free:       { capacity: 100,  refillRatePerSec: 1  },
  pro:        { capacity: 500,  refillRatePerSec: 10 },
  enterprise: { capacity: 5000, refillRatePerSec: 100 },
};

async function rateLimitMiddleware(req, res, next) {
  const plan = req.tenant?.plan ?? 'free';
  const config = PLAN_LIMITS[plan];

  const { allowed, retryAfterMs } = await tokenBucketRateLimit(req.tenant.id, config);

  if (!allowed) {
    res.setHeader('Retry-After', Math.ceil(retryAfterMs / 1000));
    res.setHeader('X-RateLimit-Limit', config.capacity);
    return res.status(429).json({
      error: 'Rate limit exceeded',
      retryAfter: Math.ceil(retryAfterMs / 1000),
    });
  }

  next();
}

// ── Distributed rate limit across N service instances ─────────────────────
// The Lua scripts above are atomic at the Redis level, so they work correctly
// across any number of Node.js instances sharing the same Redis cluster.
// For Redis Cluster mode, all keys for the same tenant must land on the same
// slot — use hash tags: key = `rl:{tenantId}:endpoint`
```

#### Do's and Don'ts

**Do's**
- Use Lua scripts to make rate limit operations atomic — a non-atomic implementation (GET, check, INCR) has race conditions under concurrent requests
- Set `Retry-After` and `X-RateLimit-*` response headers so clients can back off intelligently
- Use hash tags (`{tenantId}`) in Redis Cluster to ensure all keys for the same tenant land on the same slot
- Apply different limits at different layers (API gateway: per-IP; service level: per-tenant; endpoint level: per operation)
- Log rate limit rejections as metrics so you can tune limits based on real traffic patterns

**Don'ts**
- Don't implement rate limiting only in application code without Redis — it breaks in cluster/multi-instance deployments
- Don't use a fixed window algorithm for payment or mutation endpoints — the burst problem can allow 2× the intended limit
- Don't apply the same limit to all tenants — enterprise plans legitimately need much higher limits
- Don't forget to return 429 (not 503) for rate limiting — 503 signals a service problem, 429 signals client throttling
- Don't set limits too aggressively for legitimate retry scenarios — ensure the limit is above the expected burst for normal operation

#### Common Follow-up Questions

1. How would you implement per-endpoint rate limiting in addition to per-tenant limits, and how do the two interact?
2. What happens to your rate limiter if the Redis instance goes down? Should you fail open (allow) or fail closed (deny)?
3. How would you implement a rate limiter that applies to a group of related tenants (e.g., an agency account and all its sub-accounts)?

### **64. Capacity Planning & Load Testing**

> **Level:** Advanced

#### Why This Is Asked

HighLevel processes 4 billion API hits and 2.5 billion message events daily. At this scale, under-provisioning causes incidents; over-provisioning burns budget. A Lead Engineer must be able to forecast capacity needs, define autoscaling thresholds, run load tests, and communicate findings to SRE and infrastructure teams.

#### Explanation

**Capacity Planning Process**

1. **Measure current baselines** — p50/p95/p99 latency, RPS, CPU/memory per pod at typical load
2. **Forecast growth** — use historical traffic trends (day-of-week patterns, MoM growth rate)
3. **Define the SLO** — e.g., p99 < 500 ms at 2× current peak
4. **Load test to find the breaking point** — find RPS where p99 breaches the SLO
5. **Calculate headroom** — `max_safe_rps = breaking_point_rps × 0.7` (30% safety margin)
6. **Size the fleet** — `pods_needed = (peak_rps / max_safe_rps_per_pod) × redundancy_factor`
7. **Set autoscaling thresholds** — HPA target at 60–70% of the safe maximum

**Little's Law**

```
L = λ × W
Concurrency = Arrival rate (RPS) × Average response time (seconds)
```

Example: 1000 RPS × 0.2 s average latency = 200 concurrent requests. A pool of 200 goroutines or connections is needed.

**Node.js Memory Sizing**

```
container_memory_limit = heap_size + off_heap (buffers, native) + OS overhead
                       ≈ --max-old-space-size + 200 MB overhead
Rule of thumb: set limit to 1.5× the observed p99 RSS during load test
```

**Load Testing Tools**

| Tool | Strengths |
|---|---|
| **k6** | JS scripting, Prometheus export, cloud execution |
| **Artillery** | YAML-based, good for microservices |
| **autocannon** | Minimal, Node.js native, CI-friendly |
| **wrk** | C-based, extreme throughput, minimal scripting |

#### Code Example

```javascript
// ── k6 load test script ────────────────────────────────────────────────────
// Run: k6 run --vus 200 --duration 5m load-test.js
```

### **65. Performance Framework Design & Reusable Tooling**

> **Level:** Advanced

#### Why This Is Asked

The HighLevel JD explicitly asks for engineers who "define best practices for high-performance microservices" and "create reusable performance tooling, dashboards, and profiling workflows." This is a Lead-level expectation — you need to build systems that enable other engineers to write performant code, not just fix issues yourself.

#### Explanation

**Performance Budget**

A performance budget defines the maximum acceptable values for key metrics: p99 latency, startup time, memory RSS, event loop lag. Any PR that breaches the budget requires explicit approval or a fix.

**Regression Benchmarking in CI**

Automated benchmarks run on every PR and compare against a baseline. If a function's performance regresses by more than a threshold (e.g., 10%), the CI check fails.

**Shared Observability Library**

Rather than each microservice wiring up Prometheus, OpenTelemetry, and health check endpoints independently, a shared internal npm package provides:
- Pre-configured OTel NodeSDK setup
- Standard Prometheus metrics (HTTP, DB, event loop lag)
- Standard health check endpoints

Documented, runnable steps to profile a service in production:
1. Enable `--inspect` on a single pod behind a debug label selector
2. Connect via `kubectl port-forward` and Chrome DevTools or `0x`
3. Run a targeted load test to generate the hot path
4. Download flame graph and heap snapshot

### **66. Redis Streams & Consumer Groups**

> **Level:** Advanced

#### Why This Is Asked

HighLevel processes 2.5 billion message events daily. Redis Streams provide a durable, ordered, consumer-group-based message queue that sits between lightweight Pub/Sub (no persistence) and heavyweight Kafka. Interviewers probe whether you can choose the right tool and implement consumer groups correctly.

#### Explanation

**Redis Streams vs Pub/Sub vs Kafka**

| | Redis Pub/Sub | Redis Streams | Kafka |
|---|---|---|---|
| Persistence | None (fire-and-forget) | Yes (configurable trim) | Yes (log compaction) |
| Consumer groups | No | Yes | Yes |
| Ordering | Per-channel | Strict (by ID) | Per-partition |
| Replay | No | Yes (read from ID) | Yes |
| Throughput | Very high | High | Very high |
| Operational complexity | Low | Low | High |
| Best for | Real-time events, cache invalidation | Task queues, audit logs | High-scale event streaming |

**Key Commands**

- `XADD stream * field value` — append message (auto-generated ID)
- `XREAD COUNT n BLOCK ms STREAMS stream id` — consume messages
- `XGROUP CREATE stream group $ MKSTREAM` — create consumer group
- `XREADGROUP GROUP group consumer COUNT n BLOCK ms STREAMS stream >` — read undelivered messages
- `XACK stream group id` — acknowledge a message
- `XPENDING stream group - + n` — list unacknowledged messages

**Pending Entries List (PEL)**

Messages delivered to a consumer but not yet ACKed live in the PEL. A separate **claim worker** (`XCLAIM`) periodically scans for messages stuck in the PEL beyond a timeout and reassigns them to healthy consumers — this is how at-least-once delivery is achieved.

#### Code Example

```javascript
const redis = require('ioredis');
const client = new redis(process.env.REDIS_URL);

const STREAM = 'contact-events';
const GROUP = 'contacts-worker';
const CONSUMER = `worker-${process.pid}`;
const BLOCK_MS = 2000;   // block for 2 s waiting for new messages
const CLAIM_TIMEOUT = 30_000; // reclaim messages stuck > 30 s

// ── Producer ──────────────────────────────────────────────────────────────
async function publishContactEvent(type, payload) {
  const id = await client.xadd(
    STREAM,
    '*',                     // auto-generate ID
    'type', type,
    'payload', JSON.stringify(payload),
    'tenantId', payload.tenantId,
    'timestamp', Date.now().toString()
  );
  return id;
}

// ── Setup — create stream and consumer group ───────────────────────────────
async function setupStream() {
  try {
    // MKSTREAM creates the stream if it doesn't exist
    // '$' means start reading only new messages (not historical)
    await client.xgroup('CREATE', STREAM, GROUP, '$', 'MKSTREAM');
    console.log(`Consumer group ${GROUP} created`);
  } catch (err) {
    if (!err.message.includes('BUSYGROUP')) throw err;
    // Group already exists — that's fine
  }
}

// ── Consumer worker ────────────────────────────────────────────────────────
async function startConsumer() {
  await setupStream();
  console.log(`Consumer ${CONSUMER} started`);

  while (true) {
    try {
      // '>' means: give me messages not yet delivered to any consumer in this group
      const results = await client.xreadgroup(
        'GROUP', GROUP,
        CONSUMER,
        'COUNT', 10,
        'BLOCK', BLOCK_MS,
        'STREAMS', STREAM, '>'
      );

      if (!results) continue; // timeout — no new messages

      for (const [, messages] of results) {
        for (const [id, fields] of messages) {
          await processMessage(id, fields);
        }
      }
    } catch (err) {
      console.error('Consumer loop error', err);
      await new Promise(r => setTimeout(r, 1000));
    }
  }
}

async function processMessage(id, fields) {
  const data = {};
  for (let i = 0; i < fields.length; i += 2) {
    data[fields[i]] = fields[i + 1];
  }

  try {
    const payload = JSON.parse(data.payload);
    await handleContactEvent(data.type, payload);
    await client.xack(STREAM, GROUP, id);
  } catch (err) {
    console.error(`Failed to process message ${id}`, err);
    // Message stays in PEL — will be reclaimed after CLAIM_TIMEOUT
  }
}

// ── Claim worker — recover stuck messages ─────────────────────────────────
async function runClaimWorker() {
  while (true) {
    await new Promise(r => setTimeout(r, 10_000)); // run every 10 s

    try {
      // List messages pending for longer than CLAIM_TIMEOUT
      const pending = await client.xpending(
        STREAM, GROUP, '-', '+', 10
      );

      for (const entry of pending) {
        const [id, , idleMs] = entry;
        if (idleMs < CLAIM_TIMEOUT) continue;

        // Transfer ownership to this consumer
        const claimed = await client.xclaim(
          STREAM, GROUP, CONSUMER,
          CLAIM_TIMEOUT, id
        );

        if (claimed.length) {
          const [claimedId, fields] = claimed[0];
          console.warn(`Reclaimed stuck message ${claimedId}`);
          await processMessage(claimedId, fields);
        }
      }
    } catch (err) {
      console.error('Claim worker error', err);
    }
  }
}

// ── Stream trimming — prevent unbounded growth ────────────────────────────
// XADD stream MAXLEN ~ 100000 * field value
// '~' means approximate trimming (more efficient than exact)
async function publishWithTrim(type, payload) {
  return client.xadd(
    STREAM,
    'MAXLEN', '~', 100_000, // keep ~100k messages
    '*',
    'type', type,
    'payload', JSON.stringify(payload)
  );
}
```

#### Do's and Don'ts

**Do's**
- Always run a claim worker alongside your consumers to recover messages stuck in the PEL after a consumer crash
- Use `MAXLEN ~` (approximate trimming) to cap stream size — exact trimming is O(n) per insertion
- Acknowledge messages only after successful processing — not before
- Use multiple consumer names per group to parallelize consumption across pod instances
- Monitor PEL depth as a metric — growing PEL means messages are not being processed

**Don'ts**
- Don't use Redis Streams when you need schema evolution, complex routing, or replay across months — Kafka is better for those cases
- Don't set `BLOCK 0` (block indefinitely) in production without a health check timeout — a long-blocking client looks like a hung connection
- Don't skip the claim worker — without it, a crashed consumer's messages sit in the PEL forever
- Don't trim the stream so aggressively that you lose messages before all consumer groups have read them

1. How does Redis Streams handle backpressure compared to GCP Pub/Sub?
2. You have 5 consumer instances but one is 10× slower than the others. How does the consumer group model handle this?
3. When would you choose Redis Streams over GCP Pub/Sub for a new feature at HighLevel?

### **67. Node.js Worker Threads for CPU-Intensive Work**

> **Level:** Advanced

#### Why This Is Asked

Node.js's single-threaded event loop is a strength for I/O but a weakness for CPU-intensive work. Parsing large CSV imports, generating PDF reports, running complex aggregations, or executing WASM — all of these block the event loop. A Lead Engineer must know when and how to offload to Worker Threads without over-engineering.

#### Explanation

**Event Loop Blocking Problem**

Any synchronous JavaScript that takes more than ~1–5 ms on the main thread delays every other pending callback. In a server, this means latency spikes for all concurrent requests while one heavy operation is running.

**Worker Threads vs Child Processes vs Cluster**

| | Worker Threads | Child Process | Cluster |
|---|---|---|---|
| Memory sharing | SharedArrayBuffer | None (IPC) | None |
| Startup cost | Low (same process) | High (fork) | High (fork) |
| Communication | postMessage, SharedArrayBuffer | IPC (JSON stringify) | IPC |
| Crash isolation | None (crashes main process) | Full | Full |
| Best for | CPU-intensive, shared data | Untrusted or legacy code | HTTP server scaling |

**Worker Pool Pattern**

Creating a new Worker for each task is expensive. A worker pool maintains N pre-started workers and distributes tasks via a queue.

**SharedArrayBuffer**

Workers and the main thread can share memory via `SharedArrayBuffer`. Use `Atomics` for safe concurrent reads/writes without data races.

#### Code Example

```javascript
// ── worker-pool.js — reusable worker thread pool ──────────────────────────
const { Worker } = require('worker_threads');
const path = require('path');
const os = require('os');

class WorkerPool {
  constructor(workerScript, size = os.cpus().length - 1) {
    this._size = Math.max(1, size);
    this._workers = [];
    this._queue = [];
    this._taskId = 0;
    this._pendingTasks = new Map();

    for (let i = 0; i < this._size; i++) {
      this._spawnWorker();
    }
  }

  _spawnWorker() {
    const worker = new Worker(this._workerScript);
    const state = { worker, busy: false };

    worker.on('message', ({ id, result, error }) => {
      const task = this._pendingTasks.get(id);
      if (!task) return;
      this._pendingTasks.delete(id);
      state.busy = false;

      if (error) task.reject(new Error(error));
      else task.resolve(result);

      this._drainQueue();
    });

    worker.on('error', (err) => {
      console.error('Worker error', err);
      // Reject all tasks assigned to this worker
      this._pendingTasks.forEach((task, id) => {
        task.reject(err);
        this._pendingTasks.delete(id);
      });
      // Replace the crashed worker
      this._workers = this._workers.filter(s => s !== state);
      this._spawnWorker();
    });

    this._workers.push(state);
  }

  _drainQueue() {
    if (!this._queue.length) return;
    const idleWorker = this._workers.find(s => !s.busy);
    if (!idleWorker) return;

    const { id, payload, resolve, reject } = this._queue.shift();
    idleWorker.busy = true;
    this._pendingTasks.set(id, { resolve, reject });
    idleWorker.worker.postMessage({ id, payload });
  }

  run(payload) {
    return new Promise((resolve, reject) => {
      const id = ++this._taskId;
      const idleWorker = this._workers.find(s => !s.busy);
      if (idleWorker) {
        idleWorker.busy = true;
        this._pendingTasks.set(id, { resolve, reject });
        idleWorker.worker.postMessage({ id, payload });
      } else {
        this._queue.push({ id, payload, resolve, reject });
      }
    });
  }

  async destroy() {
    await Promise.all(this._workers.map(s => s.worker.terminate()));
  }
}

// ── csv-worker.js — the actual worker script ──────────────────────────────
const { parentPort } = require('worker_threads');
const { parse } = require('csv-parse/sync');

parentPort.on('message', ({ id, payload }) => {
  try {
    const records = parse(payload.csv, { columns: true, skip_empty_lines: true });
    const processed = records.map(row => ({
      name: row.name?.trim(),
      email: row.email?.toLowerCase().trim(),
      phone: row.phone?.replace(/\D/g, ''),
    })).filter(r => r.email);

    parentPort.postMessage({ id, result: processed });
  } catch (err) {
    parentPort.postMessage({ id, error: err.message });
  }
});

// ── Usage in Express handler ───────────────────────────────────────────────
const pool = new WorkerPool('./workers/csv-worker.js', 4);

app.post('/contacts/import', upload.single('file'), async (req, res) => {
  const csv = req.file.buffer.toString('utf8');

  try {
    // Offload CPU-intensive CSV parsing to worker pool
    const contacts = await pool.run({ csv });
    await contactService.bulkCreate(req.tenant.id, contacts);
    res.json({ imported: contacts.length });
  } catch (err) {
    res.status(422).json({ error: err.message });
  }
});

// ── SharedArrayBuffer — zero-copy data sharing ────────────────────────────
const { Worker, isMainThread, workerData, parentPort: wp } = require('worker_threads');

if (isMainThread) {
  const sharedBuffer = new SharedArrayBuffer(4); // 4 bytes = Int32
  const counter = new Int32Array(sharedBuffer);

  const worker = new Worker(__filename, { workerData: { sharedBuffer } });

  worker.on('message', () => {
    console.log('Final counter value:', Atomics.load(counter, 0));
  });
} else {
  const counter = new Int32Array(workerData.sharedBuffer);
  for (let i = 0; i < 1_000_000; i++) {
    Atomics.add(counter, 0, 1); // atomic increment — no data race
  }
  wp.postMessage('done');
}
```

#### Do's and Don'ts

**Do's**
- Use a worker pool rather than spawning a new Worker per task — worker creation is expensive
- Set pool size to `CPUs - 1` to leave one core for the event loop and I/O callbacks
- Always handle `worker.on('error')` and replace crashed workers — an unhandled worker error can propagate to the main process
- Use `worker.terminate()` in graceful shutdown to clean up worker threads before process exit
- Measure event loop lag before and after offloading to validate that the worker thread actually helps

**Don'ts**
- Don't use Worker Threads for I/O-bound work — they add overhead without benefit; use async I/O instead
- Don't transfer large objects via `postMessage` without using `Transferable` objects — serialisation of large objects is slow
- Don't share mutable state between the main thread and workers without `Atomics` — data races will corrupt your data
- Don't create a worker thread pool for tasks that take < 1 ms — the IPC overhead exceeds the computation cost

### **68. Database Connection Pooling — Deep Dive**

> **Level:** Advanced

#### Why This Is Asked

At HighLevel's scale (250+ microservices), connection pools are a hidden bottleneck. Too few connections starves throughput; too many overwhelms the database. Lead Engineers must be able to size pools correctly, detect exhaustion, implement per-tenant caps, and monitor pool health as a first-class metric.

#### Explanation

**Why Pooling**

Opening a DB connection is expensive — TCP handshake, TLS negotiation, authentication. A pool pre-opens N connections and reuses them across requests. `acquireConnection → execute → releaseConnection` is microseconds instead of milliseconds.

**Pool Sizing Formula (Little's Law)**

```
optimal_pool_size = (db_cores × 2) + effective_spindle_count
```
The Hikari (Java) rule of thumb adapted for Node.js: `min(max_concurrency, db_cores × 2 + 2)`.

For Node.js microservices with an external DB:
```
per_pod_pool_size = ceil(target_rps_per_pod × avg_query_time_seconds × safety_factor)
total_connections = per_pod_pool_size × number_of_pods

must satisfy: total_connections < db_max_connections × 0.8
```

**Connection Pool Exhaustion**

When all connections are in use, new requests wait in an acquire queue. If the queue is full or the acquire timeout fires, the request fails with "connection pool exhausted". This cascades — slow queries hold connections, starving other requests.

**MongoDB vs pg (PostgreSQL) pooling**

| | MongoDB (mongoose) | pg (node-postgres) |
|---|---|---|
| Pool | `poolSize` / `maxPoolSize` | `pg.Pool({ max })` |
| Acquire timeout | `waitQueueTimeoutMS` | `connectionTimeoutMillis` |
| Idle timeout | `socketTimeoutMS` | `idleTimeoutMillis` |
| Monitoring | `pool.stats()` | Pool events |

#### Code Example

```javascript
// ── MongoDB connection pool configuration ─────────────────────────────────
const { MongoClient } = require('mongodb');

const mongoClient = new MongoClient(process.env.MONGODB_URI, {
  maxPoolSize: 20,           // max open connections per pod
  minPoolSize: 5,            // keep at least 5 warm
  maxIdleTimeMS: 30_000,     // close connections idle > 30 s
  waitQueueTimeoutMS: 5_000, // fail after 5 s waiting for a connection
  serverSelectionTimeoutMS: 3_000,
  connectTimeoutMS: 5_000,
  socketTimeoutMS: 45_000,
});

// ── PostgreSQL pool configuration ──────────────────────────────────────────
const { Pool } = require('pg');

const pgPool = new Pool({
  connectionString: process.env.PG_URI,
  max: 20,
  min: 5,
  idleTimeoutMillis: 30_000,
  connectionTimeoutMillis: 5_000,
  statement_timeout: 10_000,    // kill queries running > 10 s
  query_timeout: 10_000,
});

pgPool.on('error', (err) => {
  console.error('PG pool idle client error', err);
});

// ── Pool health metrics ────────────────────────────────────────────────────
const client = require('prom-client');

const poolWaiting = new client.Gauge({
  name: 'db_pool_waiting_count',
  help: 'Number of requests waiting for a DB connection',
  labelNames: ['pool'],
});

const poolTotal = new client.Gauge({
  name: 'db_pool_total_connections',
  help: 'Total connections in the pool',
  labelNames: ['pool'],
});

const poolIdle = new client.Gauge({
  name: 'db_pool_idle_connections',
  help: 'Idle connections in the pool',
  labelNames: ['pool'],
});

setInterval(() => {
  // PostgreSQL pool stats
  poolTotal.set({ pool: 'pg' }, pgPool.totalCount);
  poolIdle.set({ pool: 'pg' }, pgPool.idleCount);
  poolWaiting.set({ pool: 'pg' }, pgPool.waitingCount);
}, 5000).unref();

// ── Per-tenant connection cap (pool-model multi-tenancy) ───────────────────
const tenantConcurrency = new Map(); // tenantId → active connection count

async function withTenantConnection(tenantId, fn, { maxConcurrent = 5 } = {}) {
  const current = tenantConcurrency.get(tenantId) ?? 0;

  if (current >= maxConcurrent) {
    throw Object.assign(new Error('Tenant connection limit exceeded'), {
      code: 'TENANT_THROTTLED',
      statusCode: 429,
    });
  }

  tenantConcurrency.set(tenantId, current + 1);
  try {
    return await fn();
  } finally {
    const after = (tenantConcurrency.get(tenantId) ?? 1) - 1;
    if (after <= 0) tenantConcurrency.delete(tenantId);
    else tenantConcurrency.set(tenantId, after);
  }
}

// ── Query timeout wrapper ──────────────────────────────────────────────────
async function queryWithTimeout(pool, query, params, timeoutMs = 8_000) {
  const timeoutError = new Error(`Query timed out after ${timeoutMs}ms`);
  const timeout = new Promise((_, reject) =>
    setTimeout(() => reject(timeoutError), timeoutMs)
  );
  return Promise.race([pool.query(query, params), timeout]);
}

// ── Graceful pool shutdown ─────────────────────────────────────────────────
async function shutdownPools() {
  await Promise.all([
    mongoClient.close(),
    pgPool.end(),
  ]);
}
process.on('SIGTERM', shutdownPools);
```

#### Do's and Don'ts

**Do's**
- Set `waitQueueTimeoutMS` / `connectionTimeoutMillis` — without it, a pool exhaustion causes requests to hang indefinitely
- Monitor `waitingCount` as a key health metric — a non-zero queue means you're at or near pool capacity
- Set `statement_timeout` (PostgreSQL) or use application-level query timeouts — long-running queries hold connections and starve others
- Size `maxPoolSize` based on the sizing formula and verify `total_connections × pods < db_max_connections × 0.8`
- Use `minPoolSize` > 0 so connections are pre-warmed; a cold pool under burst traffic adds latency

**Don'ts**
- Don't set `maxPoolSize` to 1 in test environments — tests will pass individually but fail under concurrent test runs
- Don't open a new MongoClient per request — each client manages its own pool; one per process is correct
- Don't ignore pool error events — silent pool errors lead to mysterious query hangs
- Don't size the pool without considering the database's `max_connections` ceiling — 100 pods × 50 connections = 5000 connections, which exceeds most default DB configurations

### **69. Observability Stack — Grafana, Prometheus & Alerting**

> **Level:** Advanced

#### Why This Is Asked

The HighLevel JD lists Grafana, Prometheus, and OpenTelemetry as required tools. Interviewers want to see that you can design dashboards that actually reveal production issues, write PromQL for burn rate alerts, and define actionable runbooks — not just that you know what the tools are.

#### Explanation

**Four Signals (Google SRE book)**

1. **Latency** — time to serve a request (p50/p95/p99)
2. **Traffic** — RPS, messages/sec
3. **Errors** — error rate (5xx, failed messages, exception count)
4. **Saturation** — CPU, memory, event loop lag, connection pool usage

**Recording Rules**

Recording rules pre-compute expensive PromQL expressions and store the result as a new metric. This makes dashboards fast and enables alerting on computed values.

**Multi-Window Burn Rate Alerts**

SLO alerting based on a single window has high false-positive or false-negative rates. The industry-standard approach uses two windows:
- **Short window** (1 h) — fast detection of severe burns
- **Long window** (6 h) — confirmation that the burn is sustained, not a spike

**Alert Routing**

Alerts should route to the right channel based on severity. Severity 1 (p1) pages on-call; Severity 2 goes to Slack #incidents; Severity 3 creates a ticket.

#### Code Example

```yaml
# ── prometheus-rules.yaml ─────────────────────────────────────────────────

groups:
  - name: nodejs-service
    interval: 30s
    rules:

      # Recording rule — pre-compute p99 latency per service/route
      - record: job:http_request_duration_seconds:p99
        expr: |
          histogram_quantile(0.99,
            sum by (job, route, le) (
              rate(http_request_duration_seconds_bucket[5m])
            )
          )

      # Recording rule — error rate per service
      - record: job:http_error_rate:rate5m
        expr: |
          sum by (job) (rate(http_request_duration_seconds_count{status_code=~"5.."}[5m]))
          /
          sum by (job) (rate(http_request_duration_seconds_count[5m]))

      # ── SLO Alerts — multi-window burn rate ──────────────────────────────

      # Burn rate > 14× over 1 h (exhausts 30-day error budget in 2 h)
      - alert: SLOErrorBudgetCritical
        expr: |
          (
            job:http_error_rate:rate1h > (14 * 0.001)  # 14× of 0.1% SLO
          ) and (
            job:http_error_rate:rate5m > (14 * 0.001)
          )
        for: 2m
        labels:
          severity: page
        annotations:
          summary: "{{ $labels.job }} burning error budget at 14× rate"
          description: "Service {{ $labels.job }} will exhaust its 30-day error budget in ~2 hours."
          runbook: "https://wiki.internal/runbooks/slo-error-budget"

      # Burn rate > 6× over 6 h (exhausts budget in ~5 days)
      - alert: SLOErrorBudgetWarning
        expr: |
          (
            job:http_error_rate:rate6h > (6 * 0.001)
          ) and (
            job:http_error_rate:rate30m > (6 * 0.001)
          )
        for: 15m
        labels:
          severity: ticket
        annotations:
          summary: "{{ $labels.job }} error budget degrading"

      # Event loop lag alert
      - alert: EventLoopLagHigh
        expr: |
          histogram_quantile(0.99,
            sum by (job, le) (
              rate(nodejs_event_loop_lag_seconds_bucket[5m])
            )
          ) > 0.1
        for: 5m
        labels:
          severity: page
        annotations:
          summary: "High event loop lag on {{ $labels.job }}"
          description: "p99 event loop lag is {{ $value | humanizeDuration }}, indicating CPU saturation or blocking code."

      # Connection pool exhaustion alert
      - alert: DBPoolExhaustion
        expr: db_pool_waiting_count{pool="pg"} > 5
        for: 2m
        labels:
          severity: page
        annotations:
          summary: "PostgreSQL connection pool near exhaustion"
```

#### Do's and Don'ts

**Do's**
- Use recording rules for any PromQL expression queried by dashboards or alerts — raw histogram queries are expensive at scale
- Design dashboards around the Four Golden Signals — they cover the most common failure modes
- Include a runbook link in every alert annotation — on-call engineers should not have to guess what to do at 3 AM
- Use multi-window burn rate alerts rather than simple threshold alerts — they dramatically reduce false positives
- Version-control your Prometheus rules and Grafana dashboards as code (YAML / Jsonnet)

**Don'ts**
- Don't alert on every metric — alert only on symptoms (user-visible degradation), not causes
- Don't set alert `for` duration to 0 — brief spikes create noisy pages; use at least 2–5 minutes for critical alerts
- Don't put raw metric queries (without recording rules) in high-frequency dashboards — they increase Prometheus query load
- Don't make dashboards with 50 panels — focus on the 8–10 metrics that matter for the service; link to more detailed dashboards for drill-down

### **70. V8 Internals & Node.js Runtime Optimisation**

> **Level:** Advanced

#### Why This Is Asked

HighLevel profiles Node.js services at the CPU and heap level. A Lead Engineer must understand *why* certain JavaScript patterns are slow — not just that they are — to write code that the V8 JIT compiler can optimise and to mentor other engineers on the same.

#### Explanation

**Hidden Classes (Shapes)**

V8 assigns each object a hidden class (internal type). When an object's shape changes (adding/removing/reordering properties), V8 creates a new hidden class. Objects that share a hidden class can share JIT-compiled code — objects with diverging shapes cannot. Polymorphic property access is slower than monomorphic.

**Inline Caching (IC)**

V8 caches the location of properties in objects. A function that always receives objects of the same hidden class hits a **monomorphic IC** — very fast. Receiving 2 different shapes = **polymorphic IC** (slower). >4 shapes = **megamorphic IC** (no caching, slow).

**TurboFan Deoptimisation**

TurboFan is V8's optimising JIT compiler. It makes assumptions about types and shapes. If those assumptions are violated at runtime, the code is **deoptimised** back to the interpreter. Frequent deoptimisation is a performance cliff.

**Common Deoptimisation Triggers**

- Changing an object's shape after creation (adding properties dynamically)
- Changing a variable's type (number → string)
- Using `arguments` object in a function
- Using `try/catch` inside a hot function (in older V8; mostly fixed in modern V8)
- Calling `delete` on object properties

**Garbage Collection**

V8 uses generational GC:
- **Young generation** (Scavenger) — short-lived objects; very fast GC
- **Old generation** (Mark-Sweep / Mark-Compact) — survived multiple GC cycles; slower, stop-the-world pauses for full GC

Minimise GC pressure by avoiding large temporary allocations in hot paths — reuse objects and buffers.

#### Code Example

```javascript
// ── Hidden class — good vs bad patterns ───────────────────────────────────

// BAD: two objects with different shapes — polymorphic IC
function processContact(contact) {
  return contact.name.toUpperCase(); // V8 sees different hidden classes
}
processContact({ name: 'Alice', email: 'a@b.com' });
processContact({ name: 'Bob', phone: '123' }); // different shape!

// GOOD: always construct objects with the same shape in the same order
class Contact {
  constructor(name, email, phone) {
    this.name = name;
    this.email = email;
    this.phone = phone; // always present, always same order
  }
}

// ── Avoid dynamic property addition ───────────────────────────────────────
// BAD: changes object shape
const obj = {};
if (condition) obj.extra = 'value'; // conditionally adds property → shape change

// GOOD: declare all properties upfront
const obj2 = { extra: condition ? 'value' : undefined };

// ── Object pooling — avoid GC pressure in hot paths ───────────────────────
class ObjectPool {
  constructor(factory, reset) {
    this._factory = factory;
    this._reset = reset;
    this._pool = [];
  }

  acquire() {
    return this._pool.length ? this._pool.pop() : this._factory();
  }

  release(obj) {
    this._reset(obj);
    this._pool.push(obj);
  }
}

const requestContextPool = new ObjectPool(
  () => ({ tenantId: null, userId: null, traceId: null, startTime: 0 }),
  (ctx) => { ctx.tenantId = null; ctx.userId = null; ctx.traceId = null; ctx.startTime = 0; }
);

// ── Buffer reuse — avoid allocating new buffers in hot paths ───────────────
class BufferPool {
  constructor(size, count = 10) {
    this._pool = Array.from({ length: count }, () => Buffer.allocUnsafe(size));
    this._size = size;
  }

  acquire() {
    return this._pool.length ? this._pool.pop() : Buffer.allocUnsafe(this._size);
  }

  release(buf) {
    if (buf.length === this._size) this._pool.push(buf);
  }
}

// ── Avoid megamorphic call sites ────────────────────────────────────────────
// BAD: function called with many different object types
function serialize(entity) {
  return JSON.stringify(entity); // entity can be Contact, Deal, Task, etc.
}
// Each entity type has a different hidden class — megamorphic IC

// GOOD: create per-type serializers — each has a monomorphic IC
const contactSerializer = (c) => JSON.stringify({ id: c._id, name: c.name, email: c.email });
const dealSerializer = (d) => JSON.stringify({ id: d._id, value: d.value, stage: d.stage });

// ── Detect deoptimisations with --trace-deopt ─────────────────────────────
// node --trace-deopt --trace-opt server.js 2>&1 | grep -E 'deoptimize|optimize'

// ── V8 heap tuning flags ───────────────────────────────────────────────────
// node \
//   --max-old-space-size=2048 \      # 2 GB old generation heap
//   --max-semi-space-size=64 \       # 64 MB young generation (default 16 MB)
//   --gc-interval=200 \              # hint frequency for minor GC
//   server.js

// ── Measure GC impact with perf_hooks ─────────────────────────────────────
const { PerformanceObserver } = require('perf_hooks');
const gcPauseHistogram = new client.Histogram({
  name: 'nodejs_gc_pause_seconds',
  help: 'GC pause durations',
  labelNames: ['kind'],
  buckets: [0.0001, 0.001, 0.005, 0.01, 0.05, 0.1, 0.5],
});

const obs = new PerformanceObserver((list) => {
  for (const entry of list.getEntries()) {
    gcPauseHistogram.observe(
      { kind: entry.detail?.kind ?? 'unknown' },
      entry.duration / 1000
    );
  }
});
obs.observe({ entryTypes: ['gc'] });
```

#### Do's and Don'ts

**Do's**
- Always construct objects with the same properties in the same order — use classes or factory functions, not ad-hoc object literals
- Use `--trace-deopt` in a staging environment to identify functions being deoptimised in your hot path
- Reuse objects and buffers in high-frequency code paths to reduce GC pressure
- Set `--max-old-space-size` explicitly — Node.js default heap is 1.5 GB on 64-bit systems, which may be too low or too high for your workload
- Monitor GC pause duration with `PerformanceObserver` — long stop-the-world pauses explain mysterious p99 latency spikes

**Don'ts**
- Don't use `delete` to remove properties from objects — it changes the hidden class; set the value to `undefined` or `null` instead
- Don't mix types in the same variable in hot loops (e.g., sometimes a number, sometimes `null`) — type polymorphism prevents JIT optimisation
- Don't add properties to objects outside their constructor — always initialise all properties in the constructor
- Don't ignore GC metrics in your observability stack — high GC frequency or long pauses are often the root cause of event loop lag spikes
